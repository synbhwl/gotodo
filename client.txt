// to say hello 
curl -X GET "http://localhost:3000/hello" -H "Content-Type: application/json"

// adding a task
curl -X POST http://localhost:3000/tasks/new -H "Content-Type: application/json" -d '{"title": "learn Go"}'

// seeing all tasks
curl -X GET "http://localhost:3000/tasks/all" -H "Content-Type: application/json"

// searching a specific task
curl -X GET "http://localhost:3000/tasks/search?title=make%20this%20bloody%20thing%20work" -H "Content-Type: application/json"
